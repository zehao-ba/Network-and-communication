file = open("Note.txt")
result=""
while 1:
    line = file.readline()
    if not line:
        break
    else:
        for letter in line:
            ascii = ord(letter)
            binary = bin(ascii)[2:]         
            result=result+binary
count = 0
finalresult = ""
for num in result:
    if num == '1':
        count=count+1
    if num == '0':
        count = 0
    finalresult =finalresult + num
    if count == 5:
        count = 0
        finalresult = finalresult +'0'   
print finalresult
